prefix brt-def: <http://brt.basisregistraties.overheid.nl/def/top10nl#>
prefix geo: <http://www.w3.org/2003/01/geo/wgs84_pos#>
prefix geof: <http://www.opengis.net/def/function/geosparql/>
select ?gebouw ?gebouwWkt
where {
  <https://gemeentegeschiedenis.nl/id/municipality/Roermond> <https://gemeentegeschiedenis.nl/def/temporalExtension>/geo:hasGeometry ?gemeenteGeo .
  ?gemeenteGeo geo:asWKT ?gemeenteWkt .
  ?gebouw geo:hasGeometry ?gebouwGeo .
  ?gebouwGeo geo:asWKT ?gebouwWkt .
  ?gebouw a brt-def:Gebouw .
  filter (geof:sfContains(?gemeenteWkt, ?gebouwWkt) && !sameTerm(?gemeenteGeo,?gebouwGeo))
}
limit 5
